        - String color = "white"
        - boolean filled
        - Date dateCreated

        + isFilled() boolean
        + setFilled(boolean filled) void
        + getDateCreated() Date
        + toString() String
        + getArea() double*
        + getPerimeter() double*
    }

        + getPerimeter() double
